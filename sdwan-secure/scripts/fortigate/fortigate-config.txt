#FOS VM Config File
config system interface
edit port1
  set alias "mgmt"
  set mode dhcp
  set allowaccess ping https ssh http snmp fgfm
  set defaultgw enable
next
edit port2
  set alias "mpls"
  set role wan
  set mode dhcp
  set allowaccess ping
  set defaultgw disable
next
edit port3
  set alias "isp"
  set role wan
  set mode dhcp
  set allowaccess ping
  set defaultgw disable
next
edit port4
  set alias "lan"
  set role lan
  set mode dhcp
  set allowaccess ping
  set defaultgw disable
  next
end
config router static
    edit 1
        set dst 172.27.40.0 255.255.255.0
        set gateway 10.20.20.254
        set device "port3"
    next
    edit 2
        set dst 172.27.40.0 255.255.255.0
        set gateway 10.30.30.1
        set device "port2"
    next
end
config firewall address
    edit "DC"
        set color 6
        set subnet 172.27.40.0 255.255.255.0
    next
end
config system virtual-wan-link
  set status enable
  set load-balance-mode source-dest-ip-based
  config members
      edit 1
          set interface "port2"
      next
      edit 2
          set interface "port3"
          set gateway 10.40.40.254
      next
  end
  config health-check
      edit "ping"
          set server "172.27.40.1"
          set failtime 3
          set members 1
          config sla
              edit 1
              next
          end
      next
  end
  config service
      edit 1
          set name "streaming"
          set mode priority
          set dst "DC"
          set health-check "ping"
          set priority-members 1 2
      next
  end
end
config firewall policy
edit 1
  set name "LAN to SDWAN"
  set srcintf "port4"
  set dstintf "virtual-wan-link"
  set srcaddr "all"
  set dstaddr "all"
  set action accept
  set schedule "always"
  set service "ALL"
  set nat enable
next
end
config system admin
    edit "admin"
        set password fortinet
    next
end
